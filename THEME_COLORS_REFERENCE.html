Implementation Examples

CSS Variables

/* Light Mode */ :root { --background: 0 0% 99%; --foreground: 217 33% 17%; --primary: 263 70% 50%; --accent: 263 80% 55%; } /* Dark Mode */ .dark { --background: 217 32% 8%; --foreground: 210 40% 98%; --primary: 263 80% 65%; --accent: 200 100% 60%; } /* Usage */ body { background-color: hsl(var(--background)); color: hsl(var(--foreground)); }

Tailwind Config

theme: { extend: { colors: { background: "hsl(var(--background))", foreground: "hsl(var(--foreground))", primary: "hsl(var(--primary))", accent: "hsl(var(--accent))", } } }

Theme Toggle (React)

import { useTheme } from 'next-themes'; export function ThemeToggle() { const { theme, setTheme } = useTheme(); return ( <button onClick={() => setTheme(theme === 'dark' ? 'light' : 'dark')}> Toggle Theme </button> ); }
